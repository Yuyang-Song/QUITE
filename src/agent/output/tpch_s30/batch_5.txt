WITH MinSupplyCost AS (
    SELECT ps_partkey, MIN(ps_supplycost) AS min_cost
    FROM partsupp
    JOIN supplier ON s_suppkey = ps_suppkey
    JOIN nation ON s_nationkey = n_nationkey
    JOIN region ON n_regionkey = r_regionkey
    WHERE r_name = 'ASIA'
    GROUP BY ps_partkey
)
